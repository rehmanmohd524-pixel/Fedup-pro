pipeline {
    agent any 
    stages {
        stage('clone repo') { 
            steps {
                sh "rm -rf teks-mavenproject"
                sh "git clone https://github.com/rehmanmohd524-pixel/Fedup-pro.git
                sh "mvn clean -f mavenproject"
            }
        }
        stage('Compile') { 
            steps {
                sh "mvn compile -f mavenproject"
            }
        }
        stage('Test') { 
            steps {
                sh "mvn test -f mavenproject"
            }
        }
        stage('package') { 
            steps {
                sh "mvn package -f mavenproject"
            }
        }
        stage('install') { 
            steps {
                sh "mvn install -f mavenproject"
            }
        }
        stage('deploy') { 
            steps {
                sh "cp /root/.m2/repository/DevOps/TestProject/1.0/TestProject-1.0.war /var/lib/tomcat10/webapps"
            }
        }
    }

}
